1.
[JOINS] List the name of each employee along with their job title and department name.

Ans select name,job_title,dept_name from employees e inner join departments d inner join jobs j where e.dept_id=d.dept_id and e.job_id=j.job_id;

2.
[SUBQUERY] Find all employees whose salary is higher than the average salary of the 'IT' department.

Ans select name,salary from employees e where e.salary>
(select avg(salary) from employees es inner join departments d where es.dept_id=d.dept_id and dept_name='IT');

3.
[GROUP BY] Calculate the total salary expenditure for each department name (not ID).

Ans select dept_name,sum(salary) as Total from employees e inner join departments d where e.dept_id=d.dept_id group by dept_name;

4.
[HAVING] Display departments that have more than 2 employees and an average salary above 70,000.

Ans select dept_name,count(name) as Emp,avg(salary) as Avg from departments d inner join employees e where 
e.dept_id=d.dept_id group by dept_name having count(name)>2 and avg(salary)>70000;

5.
[SELF JOIN] Retrieve a list of employee names and their respective manager's names.

Ans select e.name as Employee,m.name as Manager from employees e inner join employees m where e.manager_id=m.emp_id;

6.
[ORDER BY] Find the top 3 highest-paid employees and list them in descending order.

Ans select name,salary from employees order by salary desc limit 3;

7.
[JOINS + WHERE] Find all Developers working in the 'San Francisco' location.

Ans select e.name,d.location from employees e inner join departments d where e.dept_id=d.dept_id and d.location='San Francisco';

8.
[GROUP BY] Find the number of employees hired each year.

Ans select Year(hire_date) as Year,count(name) from employees group by Year order by year;

9.
[HAVING] List job titles where the maximum salary of an employee in that role is less than 100,000.

Ans select job_title,max_salary from jobs having max_salary<100000;

10.
[SUBQUERY] Find the names of employees who work in the same department as 'David'.

Ans select name,dept_name from employees e inner join departments d where 
e.dept_id=d.dept_id and e.dept_id=(select dept_id from employees where name='David');

11.
[COMPLEX FILTER] Find employees who were hired between 2018 and 2020 and earn more than 60,000.

Ans select name,hire_date,salary from employees where Year(hire_date) between 2018 and 2020 and salary>60000;

12.
[CORRELATED SUBQUERY] Find employees who earn more than the average salary of their own department.

Ans select name,dept_name,salary from employees e inner join departments d where 
e.dept_id=d.dept_id and salary>(select avg(salary) from employees where e.dept_id=dept_id);

13.
[JOINS] Find the job title and salary of the employee with ID 504.

Ans select name,job_title,salary from employees e inner join jobs j where e.job_id=j.job_id and emp_id=504;

14.
[GROUP BY] For each manager (by ID), find the minimum salary of the people reporting to them.

Ans select manager_id,min(salary) from employees group by manager_id;

15.
[AGGREGATION] Find the difference between the highest and lowest salary in the company.

Ans select (max(salary)-min(salary)) as Diff from employees;

16.
[HAVING] Identify managers (by ID) who manage more than 2 people.

Ans select manager_id,count(emp_id) as Students from employees group by manager_id having students>2;

17.
[SUBQUERY] Find the employee(s) with the third-highest salary.

Ans select name,salary from employees where salary=(select salary from employees order by salary desc limit 1 offset 2);

18.
[LEFT JOIN] List all job titles and the number of employees currently holding that job.

Ans SELECT job_title,COUNT(emp_id) as total from jobs j LEFT JOIN employees e ON j.job_id = e.job_id GROUP BY j.job_title;
